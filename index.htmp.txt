function initReveal(){

/* ================================================================
   COUNTER ANIMATION
   ================================================================ */
function initCounters(){
  const io = new IntersectionObserver(entries=>{
    entries.forEach(e=>{
      if(!e.isIntersecting) return;
      io.unobserve(e.target);
      const t=parseInt(e.target.dataset.t), d=1600, s=Date.now();
      (function tick(){
        const p=Math.min((Date.now()-s)/d,1), ease=1-Math.pow(1-p,3);
        e.target.textContent=Math.round(ease*t);
        if(p<1) requestAnimationFrame(tick); else e.target.textContent=t;
      })();
    });
  },{threshold:.5});
  document.querySelectorAll('.counter').forEach(el=>io.observe(el));
}

/* ================================================================
   HERO PARTICLE CANVAS
   ================================================================ */
function initHeroCanvas(){
  const canvas = document.getElementById('heroCanvas');
  if(!canvas) return;
  const ctx = canvas.getContext('2d');
  let W,H,pts=[];
  function resize(){ W=canvas.width=canvas.offsetWidth; H=canvas.height=canvas.offsetHeight; }
  resize(); window.addEventListener('resize',resize);
  class Pt{
    constructor(){ this.reset(); }
    reset(){ this.x=Math.random()*W; this.y=Math.random()*H; this.r=Math.random()*.8+.2; this.vx=(Math.random()-.5)*.3; this.vy=(Math.random()-.5)*.3; this.a=Math.random()*.35+.05; this.ph=Math.random()*Math.PI*2; }
    update(){ this.x+=this.vx; this.y+=this.vy; this.ph+=.013; if(this.x<0||this.x>W||this.y<0||this.y>H) this.reset(); }
    draw(){ ctx.beginPath(); ctx.arc(this.x,this.y,this.r,0,Math.PI*2); ctx.fillStyle=`rgba(200,168,75,${this.a*(.5+.5*Math.sin(this.ph))})`; ctx.fill(); }
  }
  for(let i=0;i<80;i++) pts.push(new Pt());
  (function draw(){
    requestAnimationFrame(draw);
    ctx.clearRect(0,0,W,H);
    pts.forEach((p,i)=>{
      p.update(); p.draw();
      for(let j=i+1;j<pts.length;j++){
        const dx=p.x-pts[j].x, dy=p.y-pts[j].y, d=Math.sqrt(dx*dx+dy*dy);
        if(d<100){ ctx.beginPath(); ctx.moveTo(p.x,p.y); ctx.lineTo(pts[j].x,pts[j].y); ctx.strokeStyle=`rgba(200,168,75,${.04*(1-d/100)})`; ctx.lineWidth=.5; ctx.stroke(); }
      }
    });
  })();
}

function sendChatMsg(){
  const inp=document.getElementById('chatTextInput');
  const val=inp.value.trim();
  const step=CHAT_FLOW[chatState.step];
  if(!val && !step?.optional) return;
  inp.value='';
  addUserMsg(val||'—');
  processChat(val||'');
}

function processChat(val){
  const step=CHAT_FLOW[chatState.step];
  if(step.id==='start') chatState.data.propertyType=val;
  else if(step.id==='name') chatState.data.name=val;
  else if(step.id==='phone'){
    chatState.data.phone=val;
    // 🔥 INSTANT LEAD FIRED as soon as name + phone captured
    if(!chatState.leadFired && chatState.data.name && val.replace(/\D/g,'').length>=10){
      chatState.leadFired=true;
      const instantLead={ name:chatState.data.name, phone:val, propertyType:chatState.data.propertyType, source:'chatbot-instant', timestamp:new Date().toISOString() };
      saveLead(instantLead);
      sendToWeb3Forms(instantLead);
      sendMailtoNotification(instantLead);
      // Show green confirmation bar
      const bar=document.getElementById('leadSavedBar');
      bar.style.display='flex';
      setTimeout(()=>bar.style.display='none', 5000);
    }
  }
  else if(step.id==='service') chatState.data.service=val;
  else if(step.id==='email') chatState.data.email=val;
  else if(step.id==='detail') chatState.data.message=val;

  chatState.step++;
  if(chatState.step>=CHAT_FLOW.length) completeChatFlow();
  else showChatStep(chatState.step);
}

function completeChatFlow(){
  setProgress(100);
  document.getElementById('chatInputRow').style.display='none';
  document.getElementById('chatQRs').innerHTML='';
  const n=chatState.data.name?.split(' ')[0]||'there';
  const phone=chatState.data.phone||'—';
  // Save full lead
  const fullLead={...chatState.data, source:'chatbot-complete', timestamp:new Date().toISOString()};
  saveLead(fullLead);
  sendToWeb3Forms(fullLead);
  typingThen(()=>{
    addBotMsg(`🎉 Thank you, **${n}**! You've made a smart decision.\n\n✅ Our senior engineer will call you at **${phone}** within 2 hours\n✅ Free HVAC Safety Checklist sent to your email\n✅ No obligation — just honest expert advice\n\nFor immediate assistance:\n📞 **+91 93122 64832**\n📧 ajay@aircontrols.in`);
  });
}

  addUserMessage(input);

  switch(leadData.stage){

    case 0:
      leadData.name = input;
      leadData.stage = 1;
      addBotMessage("Thanks " + input + ". What's your phone number?");
      break;

    case 1:
      leadData.phone = input;
      leadData.stage = 2;
      addBotMessage("What do you need help with?");
updateOptions([
  "AC Repair",
  "AC Servicing",
  "Installation",
  "AMC",
  "Not sure"
]);
    case 2:
  leadData.service = input;

  // 🔥 HIGH VALUE SERVICE BOOST
  if(input.toLowerCase().includes("repair")){
    addBotMessage("Got it. AC repair issues can worsen quickly if delayed.");
  }

  leadData.stage = 3;

  addBotMessage("How urgent is your requirement?");
  updateOptions([
    "Immediate (Today)",
    "Within 24 hours",
    "This week"
  ]);

  break;

    case 3:
  leadData.urgency = input;

  // 🔥 HIGH INTENT DETECTED
  if(input.toLowerCase().includes("immediate") || input === "1"){
    
    addBotMessage("⚡ This looks urgent. You can connect instantly:");

    showInstantActions(); // 👈 NEW FUNCTION

    leadData.stage = 4;
    addBotMessage("Also, tell me your location so we assign nearest engineer.");
  }
  else{
    leadData.stage = 4;
    addBotMessage("Which area are you located in?");
  }

  break;
    case 4:
      leadData.location = input;
      leadData.stage = 5;

      addBotMessage("✅ Perfect. Assigning a senior engineer near you.\n\nYou’ll receive a call shortly.");

      // 🔥 FINAL CONVERSION PUSH
      showInstantActions();

      sendLeadToWhatsApp();
      break;

    default:
      addBotMessage("Our team will contact you shortly.");
  }
}
